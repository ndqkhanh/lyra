```python
# client/client.py
class LyraClient:
    """Embedded driver for Lyra. One instance owns one (lazy) session.

    Multiple instances are safe to use concurrently; sessions are
    isolated by ``session_id``. The client never touches stdin/stdout
    of the calling process — all I/O goes through return values and
    the optional ``on_event`` callback.
    """

    def __init__(
        self,
        *,
        config_path: Path | None = None,
        repo_root: Path | None = None,
        model: str | None = None,
        fast_model: str | None = None,
        smart_model: str | None = None,
        tracing: Sequence["TracingCallback"] = (),
        on_event: Callable[[StreamEvent], None] | None = None,
    ) -> None: ...

    # primary surface
    def chat(self, request: ChatRequest) -> ChatResponse: ...
    def stream(self, request: ChatRequest) -> Iterator[StreamEvent]: ...

    # introspection
    def list_models(self) -> Sequence[str]: ...
    def list_skills(self) -> Sequence["SkillManifest"]: ...
    def list_sessions(self) -> Sequence[Mapping[str, Any]]: ...

    # lifecycle
    def reset_agent(self) -> None:
        """Invalidate the cached InteractiveSession. Forces a fresh
        boot on the next call — required after config changes."""

    def close(self) -> None:
        """Flush HIR + persist final turn snapshot."""
```

```python
"""Embedded LyraClient driver.

Delegates to lyra_cli.interactive.session.InteractiveSession with the
same defaults as the REPL but with no terminal I/O. The cached session
is keyed on (model, fast_model, smart_model, repo_root, config_path) —
mutating any one of those (or calling .reset_agent()) re-creates it.
"""
from __future__ import annotations

import time
import uuid
from decimal import Decimal
from pathlib import Path
from typing import Any, Callable, Iterator, Mapping, Sequence

from .types import ChatRequest, ChatResponse, StreamEvent

# Soft type-only imports — keep client importable even on partial envs.
from typing import TYPE_CHECKING

if TYPE_CHECKING:
    from ..tracing.base import TracingCallback
    from ..skill_pkg.manifest import SkillManifest
    from ..interactive.session import InteractiveSession


class LyraClient:
    def __init__(
        self,
        *,
        config_path: Path | None = None,
        repo_root: Path | None = None,
        model: str | None = None,
        fast_model: str | None = None,
        smart_model: str | None = None,
        tracing: "Sequence[TracingCallback]" = (),
        on_event: Callable[[StreamEvent], None] | None = None,
    ) -> None:
        self._config_path = config_path
        self._repo_root = (repo_root or Path.cwd()).resolve()
        self._model = model
        self._fast_model = fast_model
        self._smart_model = smart_model
        self._tracing = tuple(tracing)
        self._on_event = on_event
        self._cached_session: "InteractiveSession | None" = None
        self._cached_session_id: str | None = None
        self._cached_key: tuple = ()

    # ------------------------------------------------------------------
    # primary surface
    # ------------------------------------------------------------------

    def chat(self, request: ChatRequest) -> ChatResponse:
        sess = self._ensure_session(request)
        sid = self._cached_session_id or str(uuid.uuid4())
        self._cached_session_id = sid
        t0 = time.perf_counter()
        text, model, ti, to, cost = sess._chat_with_llm(  # type: ignore[attr-defined]
            request.prompt,
            mode=request.mode,
            budget_usd=request.budget_usd,
        )
        latency_ms = (time.perf_counter() - t0) * 1000.0
        # Extract turn number from session.
        turn = getattr(sess, "_turn_counter", 1)
        return ChatResponse(
            text=text or "",
            model=model or self._model or "auto",
            tokens_in=int(ti or 0),
            tokens_out=int(to or 0),
            cost_usd=Decimal(str(cost or 0)),
            latency_ms=latency_ms,
            session_id=sid,
            turn=turn,
        )

    def stream(self, request: ChatRequest) -> Iterator[StreamEvent]:
        # In v3.4 we synthesise streaming by chunking the chat() result
        # into token events. Real streaming hooks in v3.5 once provider
        # streaming surface is unified across all 10 providers.
        resp = self.chat(request)
        chunk = 32
        for i in range(0, len(resp.text), chunk):
            ev = StreamEvent(kind="token", text=resp.text[i:i + chunk])
            if self._on_event:
                self._on_event(ev)
            yield ev
        done = StreamEvent(
            kind="turn_complete",
            payload={
                "model": resp.model, "tokens_in": resp.tokens_in,
                "tokens_out": resp.tokens_out, "cost_usd": str(resp.cost_usd),
                "latency_ms": resp.latency_ms,
            },
        )
        if self._on_event:
            self._on_event(done)
        yield done

    # ------------------------------------------------------------------
    # introspection
    # ------------------------------------------------------------------

    def list_models(self) -> Sequence[str]:
        from lyra_core.providers.aliases import AliasRegistry
        return tuple(AliasRegistry().canonical_slugs())

    def list_skills(self) -> "Sequence[SkillManifest]":
        from ..skill_pkg.loader import discover_skills
        return tuple(discover_skills(self._repo_root))

    def list_sessions(self) -> Sequence[Mapping[str, Any]]:
        from ..interactive.sessions_store import SessionsStore
        store = SessionsStore(self._repo_root / ".lyra" / "sessions")
        return tuple(store.summaries())

    # ------------------------------------------------------------------
    # lifecycle
    # ------------------------------------------------------------------

    def reset_agent(self) -> None:
        if self._cached_session is not None:
            try:
                self._cached_session.close()  # type: ignore[attr-defined]
            except Exception:
                pass
        self._cached_session = None
        self._cached_session_id = None
        self._cached_key = ()

    def close(self) -> None:
        self.reset_agent()

    # ------------------------------------------------------------------
    # private
    # ------------------------------------------------------------------

    def _config_key(self, request: ChatRequest) -> tuple:
        return (
            self._config_path, self._repo_root,
            self._model, self._fast_model, self._smart_model,
            request.session_id,
        )

    def _ensure_session(self, request: ChatRequest) -> "InteractiveSession":
        key = self._config_key(request)
        if self._cached_session is not None and key == self._cached_key:
            return self._cached_session
        from ..interactive.session import InteractiveSession
        sess = InteractiveSession(
            repo_root=request.repo_root or self._repo_root,
            model=self._model or "auto",
            fast_model=self._fast_model,
            smart_model=self._smart_model,
            sessions_root=(request.repo_root or self._repo_root)
                          / ".lyra" / "sessions",
            tracing=self._tracing,
        )
        if request.session_id:
            self._cached_session_id = request.session_id
            sess.resume(request.session_id)  # type: ignore[attr-defined]
        self._cached_session = sess
        self._cached_key = key
        return sess
```

```python
from __future__ import annotations

import logging
from typing import Sequence

from .base import TracingCallback, TracingEvent

_logger = logging.getLogger(__name__)


class TracingMultiplexer:
    __slots__ = ("_callbacks",)

    def __init__(self, callbacks: Sequence[TracingCallback]) -> None:
        self._callbacks = tuple(callbacks)

    def on_event(self, event: TracingEvent) -> None:
        for cb in self._callbacks:
            try:
                cb.on_event(event)
            except Exception as exc:  # noqa: BLE001
                _logger.warning("tracing callback %r failed: %s", cb, exc)
```

```python
class LangSmithCallback:
    """Emits a LangSmith RunTree per turn_end event.

    If the langsmith package is not installed, on_event is a no-op.
    """

    def __init__(
        self, *, api_key: str, project: str = "lyra",
        endpoint: str | None = None,
    ) -> None:
        self._api_key = api_key
        self._project = project
        self._endpoint = endpoint
        self._mod = _import_langsmith()
        self._client: Any | None = None

    def on_event(self, event: TracingEvent) -> None:
        if self._mod is None:
            return
        if event.get("kind") != "turn_end":
            return
        if self._client is None:
            self._client = self._mod.Client(
                api_key=self._api_key,
                api_url=self._endpoint or "https://api.smith.langchain.com",
            )
        # langsmith.RunTree is the simplest single-shot uploader.

            run = self._mod.RunTree(
                name=f"lyra/{event.get('model', 'auto')}",
                run_type="llm",
                project_name=self._project,
                inputs={"prompt": event.get("prompt", "")},
                outputs={"response": event.get("response", "")},
                extra={
                    "tokens_in": event.get("tokens_in", 0),
                    "tokens_out": event.get("tokens_out", 0),
                    "cost_usd": event.get("cost_usd", "0"),
                    "latency_ms": event.get("latency_ms", 0.0),
                    "session_id": event.get("session_id", ""),
```

```python
"""Minimal YAML frontmatter parser. Stdlib-only — supports the subset
used by SKILL.md (scalars, lists of strings, nested mappings up to one
level). Avoids adding PyYAML as a hard dep."""
from __future__ import annotations

import re
from typing import Any

_FENCE = re.compile(r"^---\s*$", re.M)


def parse_frontmatter(text: str) -> tuple[dict[str, Any], str]:
    """Return (frontmatter_dict, body). Empty dict when no fence."""
    if not text.startswith("---"):
        return {}, text
    parts = _FENCE.split(text, 2)
    if len(parts) < 3:
        return {}, text
    _, raw_fm, body = parts
    return _parse_yaml(raw_fm.strip()), body.lstrip("\n")


def _parse_yaml(text: str) -> dict[str, Any]:
    """Tiny YAML subset parser:
      key: value         -> str
      key: 'quoted'      -> str (strip quotes)
      key: 123           -> int
      key:
        - item1          -> list[str]
        - item2
      key: []            -> list (empty)
    Comments (#...) are stripped.
    """
    out: dict[str, Any] = {}
    lines = [ln.rstrip() for ln in text.splitlines()]
    i = 0
    while i < len(lines):
        ln = lines[i]
        if not ln or ln.lstrip().startswith("#"):
            i += 1
            continue
        if ":" not in ln:
            i += 1
            continue
        key, _, raw = ln.partition(":")
        key = key.strip()
        raw = raw.split("#", 1)[0].strip()
        if raw == "":
            # multi-line list
            items: list[str] = []
            j = i + 1
            while j < len(lines) and lines[j].lstrip().startswith("-"):
                items.append(_unquote(lines[j].lstrip()[1:].strip()))
                j += 1
            out[key] = items
            i = j
            continue
        if raw == "[]":
            out[key] = []
        elif raw.startswith("[") and raw.endswith("]"):
            inner = raw[1:-1]
            out[key] = [_unquote(p.strip()) for p in inner.split(",") if p.strip()]
        elif raw.lstrip("-").isdigit():
            out[key] = int(raw)
        else:
            out[key] = _unquote(raw)
        i += 1
    return out


def _unquote(value: str) -> str:
    if len(value) >= 2 and value[0] == value[-1] and value[0] in ("'", '"'):
        return value[1:-1]
    return value
```

```python
    if not _ID_RE.match(sid):
        raise SkillManifestError(f"invalid id {sid!r}")

    version = str(fm.get("version", "")).strip()
    if not _SEMVER_RE.match(version):
        raise SkillManifestError(f"invalid version {version!r}")

    name = str(fm.get("name", "")).strip()
    if not name:
        raise SkillManifestError("name is required")

    description = str(fm.get("description", "")).strip()
    if not description:
        raise SkillManifestError("description is required")

    priority = int(fm.get("priority", 50))
    if not 0 <= priority <= 100:
        raise SkillManifestError("priority must be in [0..100]")

    compat = str(fm.get("compatibility", ">=3.4.0")).strip()
    _check_compat(compat)

    requires = tuple(fm.get("requires") or ())
    triggers = tuple(fm.get("triggers") or ())

    return SkillManifest(
        id=sid, name=name, version=version, description=description,
        author=str(fm.get("author", "")).strip(),
        compatibility=compat, requires=requires,
        priority=priority, triggers=triggers,
        body=body, source_path=source,
    )


def _check_compat(spec: str) -> None:
    """Lightweight PEP 440-ish check against the running Lyra version."""
    from lyra_cli import __version__
    cur = tuple(int(x) for x in __version__.split(".")[:3])
    op_match = re.match(r"^([<>=!]+)\s*(\d+\.\d+\.\d+)$", spec.strip())
    if not op_match:
        raise SkillManifestError(f"unsupported compatibility spec {spec!r}")
    op, ver = op_match.groups()
    target = tuple(int(x) for x in ver.split("."))
    pred = {
        ">=": cur >= target, ">": cur > target,
        "<=": cur <= target, "<": cur < target,
        "==": cur == target, "!=": cur != target,
    }.get(op)
    if not pred:
        raise SkillManifestError(
            f"current Lyra {__version__} does not satisfy {spec!r}"
        )
```

```python
"""Skill installer — local path / git URL / tarball.

Local path:    /abs/or/rel/path     → copytree into dest/<id>/
Git URL:       git+https://...      → clone into tmp, then local install
Tarball URL:   https://...tar.gz    → download to tmp, extract, install

Atomicity: install is "stage to tmp dir → fsync → os.replace()", so a
partial install never corrupts an existing skill dir.
"""
from __future__ import annotations

import shutil
import subprocess
import tarfile
import tempfile
import urllib.request
from pathlib import Path

from .manifest import SkillManifest, SkillManifestError, parse_manifest


def install_skill(
    source: str, *, dest_root: Path, overwrite: bool = False,
) -> SkillManifest:
    dest_root.mkdir(parents=True, exist_ok=True)
    with tempfile.TemporaryDirectory() as tmp_str:
        tmp = Path(tmp_str)
        staged = _resolve_source(source, tmp)
        skill_md = staged / "SKILL.md"
        if not skill_md.exists():
            raise SkillManifestError(f"no SKILL.md in {source}")
        manifest = parse_manifest(skill_md.read_text(), source=skill_md)

        target = dest_root / manifest.id
        if target.exists():
            if not overwrite:
                raise FileExistsError(f"skill {manifest.id!r} already installed at {target}")
            shutil.rmtree(target)
        shutil.copytree(staged, target)

    return manifest


def _resolve_source(source: str, tmp: Path) -> Path:
    if source.startswith("git+"):
        url = source[4:]
        clone = tmp / "clone"
        subprocess.run(["git", "clone", "--depth", "1", url, str(clone)],
                       check=True, capture_output=True)
        return clone
    if source.startswith(("http://", "https://")) and source.endswith((".tar.gz", ".tgz")):
        archive = tmp / "skill.tar.gz"
        urllib.request.urlretrieve(source, archive)
        extract = tmp / "extract"
        extract.mkdir()
        with tarfile.open(archive) as tf:
            tf.extractall(extract)
        # Take the first directory at the top level.
        children = [p for p in extract.iterdir() if p.is_dir()]
        return children[0] if children else extract
    p = Path(source).expanduser().resolve()
    if not p.exists():
        raise FileNotFoundError(source)
    return p
```

```python
from __future__ import annotations

from abc import ABC, abstractmethod
from dataclasses import dataclass, field
from typing import Literal, Mapping, Sequence


@dataclass(frozen=True)
class ExecResult:
    stdout: str
    stderr: str
    exit_code: int
    elapsed_ms: float
    truncated: bool = False
    timed_out: bool = False


@dataclass(frozen=True)
class SandboxConfig:
    image: str = "python:3.11-slim"
    cpu_limit: float = 1.0
    memory_limit_mb: int = 512
    network: Literal["none", "bridge"] = "none"
    mount_repo: bool = True
    tmpfs_tmp: bool = True
    workdir: str = "/workspace"
    extra_env: Mapping[str, str] = field(default_factory=dict)


class SandboxProvider(ABC):
    @abstractmethod
    def exec(
        self, cmd: Sequence[str], *,
        cwd: str | None = None,
        env: Mapping[str, str] | None = None,
        timeout_s: float = 60.0,
        max_output_bytes: int = 1024 * 1024,
    ) -> ExecResult: ...

    @abstractmethod
    def write_file(self, path: str, content: str | bytes) -> None: ...

    @abstractmethod
    def read_file(self, path: str) -> str: ...

    @abstractmethod
    def close(self) -> None: ...
```

```python
"""Local subprocess fallback. NO ISOLATION — use only when Docker is
unavailable. Resource caps are best-effort (resource module on Unix)."""
from __future__ import annotations

import os
import subprocess
import time
from pathlib import Path
from typing import Mapping, Sequence

from .base import ExecResult, SandboxConfig, SandboxProvider


class LocalSandboxProvider(SandboxProvider):
    def __init__(self, config: SandboxConfig) -> None:
        self._cfg = config
        self._cwd = Path.cwd()

    def exec(
        self, cmd: Sequence[str], *,
        cwd: str | None = None,
        env: Mapping[str, str] | None = None,
        timeout_s: float = 60.0,
        max_output_bytes: int = 1024 * 1024,
    ) -> ExecResult:
        merged_env = {**os.environ, **(env or {}), **self._cfg.extra_env}
        t0 = time.perf_counter()
        try:
            proc = subprocess.run(
                list(cmd), cwd=cwd or str(self._cwd),
                env=merged_env, capture_output=True, text=True,
                timeout=timeout_s,
            )
        except subprocess.TimeoutExpired as exc:
            elapsed_ms = (time.perf_counter() - t0) * 1000.0
            return ExecResult(
                stdout=(exc.stdout or "")[:max_output_bytes],
                stderr=(exc.stderr or "")[:max_output_bytes],
                exit_code=124, elapsed_ms=elapsed_ms,
                truncated=False, timed_out=True,
            )
        elapsed_ms = (time.perf_counter() - t0) * 1000.0
        out = proc.stdout or ""; err = proc.stderr or ""
        truncated = (len(out) > max_output_bytes
                     or len(err) > max_output_bytes)
        if len(out) > max_output_bytes:
            out = out[:max_output_bytes] + "\n[…truncated…]\n"
        if len(err) > max_output_bytes:
            err = err[:max_output_bytes] + "\n[…truncated…]\n"
        return ExecResult(
            stdout=out, stderr=err, exit_code=proc.returncode,
            elapsed_ms=elapsed_ms, truncated=truncated, timed_out=False,
        )

    def write_file(self, path: str, content: str | bytes) -> None:
        target = (self._cwd / path).resolve()
        target.parent.mkdir(parents=True, exist_ok=True)
        if isinstance(content, bytes):
            target.write_bytes(content)
        else:
            target.write_text(content)

    def read_file(self, path: str) -> str:
        return (self._cwd / path).read_text()

    def close(self) -> None:
        return  # nothing to clean
```

```python
"""Docker-backed sandbox. Uses subprocess to drive `docker run` so we
avoid adding the `docker` python SDK as a hard dep. Resource caps go
through standard `--cpus`, `--memory`, `--network`, `--tmpfs` flags."""
from __future__ import annotations

import os
import shlex
import shutil
import subprocess
import tempfile
import time
import uuid
from pathlib import Path
from typing import Mapping, Sequence

from .base import ExecResult, SandboxConfig, SandboxProvider


class DockerSandboxProvider(SandboxProvider):
    def __init__(self, config: SandboxConfig) -> None:
        if not shutil.which("docker"):
            raise RuntimeError("docker not on PATH; use LocalSandboxProvider")
        self._cfg = config
        self._container = f"lyra-sb-{uuid.uuid4().hex[:8]}"
        self._scratch = Path(tempfile.mkdtemp(prefix="lyra-sb-"))
        self._started = False

    def _start(self) -> None:
        if self._started:
            return
        cmd = ["docker", "run", "-d", "--rm",
               "--name", self._container,
               "--cpus", str(self._cfg.cpu_limit),
               "--memory", f"{self._cfg.memory_limit_mb}m",
               "--network", self._cfg.network]
        if self._cfg.tmpfs_tmp:
            cmd += ["--tmpfs", "/tmp:size=64m"]
        if self._cfg.mount_repo:
            cmd += ["-v", f"{self._scratch}:{self._cfg.workdir}"]
        for k, v in self._cfg.extra_env.items():
            cmd += ["-e", f"{k}={v}"]
        cmd += ["-w", self._cfg.workdir, self._cfg.image,
                "tail", "-f", "/dev/null"]
        subprocess.run(cmd, check=True, capture_output=True)
        self._started = True

    def exec(
        self, cmd: Sequence[str], *,
        cwd: str | None = None,
        env: Mapping[str, str] | None = None,
        timeout_s: float = 60.0,
        max_output_bytes: int = 1024 * 1024,
    ) -> ExecResult:
        self._start()
        env_args = []
        for k, v in (env or {}).items():
            env_args += ["-e", f"{k}={v}"]
        full = (
            ["docker", "exec"] + env_args +
            (["-w", cwd] if cwd else []) +
            [self._container] + list(cmd)
        )
        t0 = time.perf_counter()
        try:
            proc = subprocess.run(
                full, capture_output=True, text=True, timeout=timeout_s,
            )
        except subprocess.TimeoutExpired as exc:
            return ExecResult(
                stdout=(exc.stdout or "")[:max_output_bytes],
                stderr=(exc.stderr or "")[:max_output_bytes],
                exit_code=124,
                elapsed_ms=(time.perf_counter() - t0) * 1000.0,
                truncated=False, timed_out=True,
            )
        elapsed_ms = (time.perf_counter() - t0) * 1000.0
        out = proc.stdout or ""; err = proc.stderr or ""
        truncated = len(out) > max_output_bytes or len(err) > max_output_bytes
        if len(out) > max_output_bytes:
            out = out[:max_output_bytes] + "\n[…truncated…]\n"
        if len(err) > max_output_bytes:
            err = err[:max_output_bytes] + "\n[…truncated…]\n"
        return ExecResult(
            stdout=out, stderr=err, exit_code=proc.returncode,
            elapsed_ms=elapsed_ms, truncated=truncated, timed_out=False,
        )

    def write_file(self, path: str, content: str | bytes) -> None:
        target = self._scratch / path
        target.parent.mkdir(parents=True, exist_ok=True)
        if isinstance(content, bytes):
            target.write_bytes(content)
        else:
            target.write_text(content)

    def read_file(self, path: str) -> str:
        return (self._scratch / path).read_text()

    def close(self) -> None:
        if self._started:
            subprocess.run(
                ["docker", "rm", "-f", self._container],
                capture_output=True,
            )
            self._started = False
        try:
            shutil.rmtree(self._scratch, ignore_errors=True)
        except Exception:
            pass
```

```python
            self._cached = None
```

```python
"""Bearer token + rate limit. In-memory only — single-process server."""
from __future__ import annotations

import time
from collections import defaultdict, deque
from typing import Deque


class RateLimiter:
    def __init__(self, per_min: int) -> None:
        self._per_min = per_min
        self._hits: dict[str, Deque[float]] = defaultdict(deque)

    def allow(self, key: str) -> bool:
        now = time.time()
        q = self._hits[key]
        while q and now - q[0] > 60.0:
            q.popleft()
        if len(q) >= self._per_min:
            return False
        q.append(now)
        return True


def check_token(env: dict, expected: str) -> tuple[bool, int]:
    """Return (ok, status_code). 401 = missing, 403 = wrong."""
    raw = env.get("HTTP_AUTHORIZATION", "")
    if not raw:
        return False, 401
    if not raw.startswith("Bearer "):
        return False, 401
    if raw[7:] != expected:
        return False, 403
    return True, 200
```

```python
from __future__ import annotations

import json
from pathlib import Path
from typing import Callable, Iterable

from lyra_cli import __version__

from ..client import ChatRequest, LyraClient
from .auth import RateLimiter, check_token

# Routes that bypass auth.
_PUBLIC_ROUTES = {"/v1/health"}


def make_wsgi_app(
    *, repo_root: Path, token: str, rate_limit_per_min: int = 60,
) -> Callable[[dict, Callable], Iterable[bytes]]:
    rl = RateLimiter(rate_limit_per_min)
    client = LyraClient(repo_root=repo_root)

    def app(environ: dict, start_response: Callable) -> Iterable[bytes]:
        path = environ.get("PATH_INFO", "")
        method = environ.get("REQUEST_METHOD", "GET").upper()

        # Auth except public routes.
        if path not in _PUBLIC_ROUTES:
            ok, status = check_token(environ, token)
            if not ok:
                return _send(start_response, status,
                             {"error": "unauthorized"})

        # Rate limit by remote addr.
        remote = environ.get("REMOTE_ADDR", "?")
        if path not in _PUBLIC_ROUTES and not rl.allow(remote):
            return _send(start_response, 429,
                         {"error": "rate_limited"})

        if path == "/v1/health" and method == "GET":
            return _send(start_response, 200,
                         {"status": "ok", "version": __version__})
        if path == "/v1/models" and method == "GET":
            return _send(start_response, 200,
                         {"data": [{"id": s} for s in client.list_models()]})
        if path == "/v1/skills" and method == "GET":
            return _send(start_response, 200,
                         {"data": [_skill_to_dict(s) for s in client.list_skills()]})
        if path == "/v1/sessions" and method == "GET":
            return _send(start_response, 200,
                         {"data": list(client.list_sessions())})
        if path == "/v1/chat" and method == "POST":
            return _chat(start_response, environ, client)
        if path == "/v1/chat/completions" and method == "POST":
            return _chat_completions(start_response, environ, client)
        if path == "/v1/stream" and method == "POST":
            return _stream(start_response, environ, client)

        return _send(start_response, 404, {"error": "not_found"})

    return app


def _send(start_response, status: int, body: dict) -> Iterable[bytes]:
    payload = json.dumps(body).encode()
    start_response(f"{status} {_status_text(status)}",
                   [("Content-Type", "application/json"),
                    ("Content-Length", str(len(payload)))])
    return [payload]


def _status_text(status: int) -> str:
    return {200: "OK", 400: "Bad Request", 401: "Unauthorized",

    try:
        body = _read_body(environ)
    except _BadJSON as exc:
        return _send(start_response, 400, {"error": "invalid_json", "detail": str(exc)})

        return _send(start_response, 400, {"error": "missing_prompt"})
```

```python
"""Progressive skill loader.

* :func:`discover_skills` walks ``<repo>/.lyra/skills/*/SKILL.md`` and
  yields parsed :class:`SkillManifest` in priority-desc order. Bad
  manifests are logged to HIR and skipped.
* :func:`load_progressive` picks the prefix that fits in a byte
  budget, with optional trigger boosting and deterministic tie-break.
"""
from __future__ import annotations

import logging
from pathlib import Path
from typing import Iterator, Sequence

from .manifest import SkillManifest, SkillManifestError, parse_manifest

_logger = logging.getLogger(__name__)


def discover_skills(repo_root: Path) -> Iterator[SkillManifest]:
    skills_dir = repo_root / ".lyra" / "skills"
    if not skills_dir.exists():
        return iter(())
    found: list[SkillManifest] = []
    for entry in sorted(skills_dir.iterdir()):
        if not entry.is_dir():
            continue
        skill_md = entry / "SKILL.md"
        if not skill_md.exists():
            continue
        try:
            found.append(parse_manifest(skill_md.read_text(),
                                         source=skill_md))
        except SkillManifestError as exc:
            _logger.debug("skipping invalid skill %s: %s", entry.name, exc)
    found.sort(key=lambda s: (-s.priority, s.id))
    return iter(found)


def load_progressive(
    skills: Sequence[SkillManifest], *,
    max_bytes: int = 16_000,
    triggers: Sequence[str] = (),
) -> tuple[SkillManifest, ...]:
    if max_bytes <= 0 or not skills:
        return ()
    trigger_set = {t.lower() for t in triggers}

    def score(s: SkillManifest) -> tuple[int, int, str]:
        boost = 1 if (trigger_set & {t.lower() for t in s.triggers}) else 0
        return (-boost, -s.priority, s.id)

    ordered = sorted(skills, key=score)
    budget = max_bytes
    picked: list[SkillManifest] = []
    for s in ordered:
        size = len(s.body.encode("utf-8")) + 200  # +overhead for header
        if size <= budget:
            picked.append(s)
            budget -= size
    return tuple(picked)
```

```python
"""Config schema versioning + import-string class loader."""
from __future__ import annotations

import importlib
from dataclasses import dataclass
from typing import Callable


CURRENT_VERSION: int = 3


@dataclass(frozen=True)
class Migration:
    from_v: int
    to_v: int
    apply: Callable[[dict], dict]


def _migrate_1_to_2(data: dict) -> dict:
    """Original v1: implicit. v2 introduced fast_model / smart_model."""
    out = dict(data)
    out.setdefault("fast_model", out.get("model", ""))
    out.setdefault("smart_model", out.get("model", ""))
    out["config_version"] = 2
    return out


def _migrate_2_to_3(data: dict) -> dict:
    """v3 adds config_version stamping + tracing block."""
    out = dict(data)
    out.setdefault("tracing", {})  # empty by default
    out["config_version"] = 3
    return out


_REGISTRY: tuple[Migration, ...] = (
    Migration(1, 2, _migrate_1_to_2),
    Migration(2, 3, _migrate_2_to_3),
)


def migrate_config(data: dict) -> dict:
    out = dict(data)
    cur = int(out.get("config_version", 1))
    if cur > CURRENT_VERSION:
        raise ValueError(
            f"config_version={cur} is newer than supported "
            f"({CURRENT_VERSION}); upgrade Lyra"
        )
    while cur < CURRENT_VERSION:
        step = next((m for m in _REGISTRY if m.from_v == cur), None)
        if step is None:
            raise ValueError(f"no migration from {cur}")
        out = step.apply(out)
        cur = step.to_v
    return out


def load_provider(spec: str) -> type:
    """Resolve ``package.module:Class`` to the class object."""
    if ":" not in spec:
        raise ValueError(f"expected 'pkg.mod:Cls', got {spec!r}")
    module_path, _, attr = spec.partition(":")
    module = importlib.import_module(module_path)
    if not hasattr(module, attr):
        raise AttributeError(f"{module_path} has no attribute {attr}")
    return getattr(module, attr)
```

## Skipped — explicit non-goals

The following DeerFlow features are **deliberately** out of scope for v3.4 and tracked for later:

| Skipped | Why | Tracked in |
|---|---|---|
| Adopt LangGraph as the runtime | Lyra has its own coroutine-based loop with first-class TDD/HIR/cron/subagent integration; swapping the substrate would be a months-long rewrite for diminishing return. | Won't fix |
| Next.js web UI | Lyra is CLI-first; web UI needs frontend dep stack + a separate distribution target. `lyra serve` already enables Claude Code / Cursor / any OpenAI-compatible UI to drive Lyra. | Won't fix unless requested |
| K8s sandbox provisioner | Docker covers the dev case; K8s adds operational burden + cluster setup. | v3.5 maybe |
| Multi-deliverable skills (skill returns several artifacts) | Current single-`SKILL.md`-as-system-prompt model has not yet hit the limit. Add only when a skill author requests it. | v3.5 maybe |
| Telegram / Slack / Feishu / WeChat IM channels | Lyra already ships 20+ adapters in `lyra_cli/channels/`; DeerFlow's are not richer. | Won't fix (no gap) |
| Cross-tool data ingestion (Phase M's roadmap) | Phase N is about embedding Lyra elsewhere, not ingesting from elsewhere. | v3.5 (per Phase M plan) |
| MCP server marketplace | Lyra already has full MCP plumbing in `lyra_cli/commands/mcp.py`. | Won't fix (no gap) |
